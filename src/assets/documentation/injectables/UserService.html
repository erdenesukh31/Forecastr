import { Injectable } from '@angular/core';
import { BehaviorSubject } from 'rxjs';
import { User } from '../../interfaces/user';
import { HttpClient } from '@angular/common/http';
import { BusinessOperationsService } from '../../shared/business-operations.service';
import { MatSnackBar } from '@angular/material';
import { PageStateService } from '../../shared/page-state.service';
import { Role } from '../../interfaces/role';
import { Level } from '../../interfaces/level';

/**
 * user service
 */
@Injectable({
  providedIn: 'root',
})
export class UserService {
  /**
   * observable which contains the current user
   */
  user$: BehaviorSubject<User>;

  /**
   * Observable which contains all team members
   */
  team$: BehaviorSubject<User[]>;

  allUsers$: BehaviorSubject<User[]>;
  
  roles$: BehaviorSubject<Role[]>;

  levels$: BehaviorSubject<Level[]>;

  /**
   * user service constructor
   * 
   * @param http
   * @param BO
   */
  constructor(
    private http: HttpClient,
    private snackBar: MatSnackBar,
    private BO: BusinessOperationsService,
    private pageState: PageStateService,
  ) {
    this.user$ = new BehaviorSubject(new User());
    this.team$ = new BehaviorSubject([]);
    this.allUsers$ = new BehaviorSubject([]);
    this.roles$ = new BehaviorSubject([]);
    this.levels$ = new BehaviorSubject([]);
  }

  /**
   * Loads the current user data from the server
   */
  initUser(): Promise<void> {
    return new Promise<void>((resolve, reject) => {
      this.http.get<User>(this.BO.getOwnUser())
        .subscribe((user: User) => {
          this.user$.next(user);
  
          if (this.team$.getValue().length === 0) {
            this.team$.next([user]);
          }
          resolve();
        }, () => reject());
    });
  }

  initializeAllUser(): void {
    this.http.get<User[]>(this.BO.getUsers())
    .subscribe((user: User[]) => {
      this.allUsers$.next(user.sort((a, b) => (a.lastName > b.lastName) ? 1 : -1));
    });
  }

  /**
   * Requests userdata from server
   */
  initializeTeam(): void {
    this.http.get<User[]>(this.BO.getTeam())
      .subscribe((user: User[]) => {
        this.team$.next(user.sort((a, b) => (a.lastName > b.lastName) ? 1 : -1));
      });
  }

  /**
   * Requests role data from server
   */
  initializeRoles(): void {
    this.http.get<any[]>(this.BO.getRoles())
      .subscribe((roles: any[]) => {
        this.roles$.next(roles);
      });
  }

  /**
   * Requests level data from server
   */
  initializeLevels(): void {

    this.levels$.next([
      {
        levelId: 2,
        name: 'Consultant',
      },
    ]);
    
/* TODO: add to backend
    this.http.get<any[]>(this.BO.getLevels())
      .subscribe((levels: any[]) => {
        this.levels$.next(levels);
      });
*/
  }

  /**
   * Empties userdata
   * Typically called at logout
   */
  reset(): void {
    this.team$.next([]);
    this.user$.next(new User());
  }

  /**
   * returns user by a given id
   */
  getUser(id: number): User | undefined {
    return this.team$.getValue().find((u: User) => u.id === id);
  }

  addUser(user: User): void {
    let users: User[] = this.allUsers$.getValue();
    users = users.filter((u: User) => u.id !== user.id);
    users.push(user);

    this.allUsers$.next(users);
  }

  editUser(user: User): void {
    let users: User[] = this.allUsers$.getValue();
    users.filter((u: User) => u.id === user.id).forEach((u: User) => {
      u.globalId = user.globalId;
      u.firstName = user.firstName;
      u.lastName = user.lastName;
      u.email = user.email;
      u.roleId = user.roleId;
      u.levelId = user.levelId;
      u.parentId = user.parentId;
      u.fte = user.fte;
      u.admin = user.admin;
    });

    this.allUsers$.next(users);
  }

  setUser(user: User): void {
    if (user.id) {
      this.http.put(this.BO.updateUser(user.id), user)
      .subscribe((u: User) => {
        this.editUser(u);
        this.snackBar.open('User successfully saved!', 'OK', { duration: 5000, });
        this.pageState.hideSpinner();

      }, (e: any) => {
        this.snackBar.open('User could not be saved!', 'OK', { duration: 10000, });
        this.pageState.hideSpinner();

      });
    } else {
      this.http.post(this.BO.createUser(), user)
      .subscribe((u: User) => {
        this.addUser(u);
        this.snackBar.open('User successfully added!', 'OK', { duration: 5000, });
        this.pageState.hideSpinner();

      }, (e: any) => {
        this.snackBar.open('User could not be added!', 'OK', { duration: 10000, });
        this.pageState.hideSpinner();

      });
    }
  }

  setUserActiveState(id: number, state: boolean): void {
    this.http.put(this.BO.setUserState(id), {id: id, active: state })
      .subscribe(() => {
        let users: User[] = this.allUsers$.getValue();
        users.filter((u: User) => u.id === id).forEach((u: User) => {
          u.active = state;
        });

        this.allUsers$.next(users);
      });
  }
}
